/* Fonction choix */
function choix(select){
    // console.log("choix() : " + select);

    let domContact = document.getElementById("reseaux");
    let domConnexion = document.getElementById("formulaire");
    let domMentionLegal = document.getElementById("texte");

    // Suppression du contact affiché de base
    if(connIsOpen == true && firstOpen == true){
        domContact.remove();
        connIsOpen = false;
        firstOpen = false;
    }

    if (select == "cont") {
        // console.log("Contact");
        if (contIsOpen) {
            // Fermer
            domContact.remove();
            contIsOpen = false;
        } else {
            // Fermer les autres
            if (connIsOpen) {
                domConnexion.remove();
                connIsOpen = false;
            }
            if (meleIsOpen) {
                domMentionLegal.remove();
                meleIsOpen = false;
            }
            // Ouvrir
            cont();
            contIsOpen = true;
        }
    } else if (select == "conn") {
        // console.log("Connexion");
        if (connIsOpen) {
            domConnexion.remove();
            connIsOpen = false;
        } else {
            if (contIsOpen) {
                domContact.remove();
                contIsOpen = false;
            }
            if (meleIsOpen) {
                domMentionLegal.remove();
                meleIsOpen = false;
            }
            conn();
            connIsOpen = true;
        }
    } else if (select == "mele") {
        // console.log("Mentions légales");
        if (meleIsOpen) {
            domMentionLegal.remove();
            meleIsOpen = false;
        } else {
            if (contIsOpen) {
                domContact.remove();
                contIsOpen = false;
            }
            if (connIsOpen) {
                domConnexion.remove();
                connIsOpen = false;
            }
            mele();
            meleIsOpen = true;
        }
    }
}
